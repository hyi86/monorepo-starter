# Code Highlight in MDX

### 단어 하이라이트

```tsx
// [!code word:onClick]
import React from 'react';

export default function Button({ label }: { label: string }) {
  return (
    <button onClick={() => alert('Clicked')} className="rounded bg-blue-500 px-4 py-2 text-white">
      {label}
    </button>
  );
}
```

### 라인 번호 표시

```ts
function add(a: number, b: number) {
  // [!code error:1]
  return a + b; // Error this line
}

function minus(a: number, b: number) {
  // [!code warning:1]
  return a - b; // Warning this line
}
```
