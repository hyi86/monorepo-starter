## Tailwind CSS Variables

<Source
  language="css"
  code={Object.keys(colors).map((key) => {
    const color = colors[key];

    const values = Object.keys(color).map((colorKey) => {
      if (key === 'mono') {
        return `--color-${colorKey}: ${color[colorKey]};`
      }
      return `--color-${key}-${colorKey}: ${color[colorKey]};`
    });

    return values.join('\n');

}).join('\n\n')}
/>
